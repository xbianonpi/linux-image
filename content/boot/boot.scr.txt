setenv fstype btrfs

setenv baseconfig ahci_imx.hotplug=1 raid=noautodetect telnet root=/dev/mmcblk0p2 video=mxcfb0:dev=hdmi,1920x1080M@60,if=RGB24 rootwait consoleblank=0 selinux=0 nohdparm splash

if test ${fstype} != btrfs; then
    setenv bootargs $baseconfig rootfstype=ext4 rootflags=data=writeback  --startup-event mountall
else
    setenv bootargs $baseconfig rootfstype=btrfs rootflags=subvol=root/@,autodefrag,compress=lz4 --startup-event mountall
fi

if test ${debug} != ""; then exit 1; fi

load mmc 0:1 ${loadaddr} zImage

if test ${board} = mx6-cubox-i; then
    if test ${cpu} = 6SOLO || test ${cpu} = 6DL; then 
        load mmc 0:1 ${fdt_addr} imx6dl-cubox-i.dtb
    else 
        load mmc 0:1 ${fdt_addr} imx6q-cubox-i.dtb
    fi
else
    load mmc 0:1 ${fdt_addr} imx6dl-hummingboard.dtb
fi

if load mmc 0:1 0x14700000 initramfs.gz; then
    setenv initram 0x14700000
else
    setenv initram "-"
fi

if test ${debug} != ""; then exit 1; fi

bootz ${loadaddr} ${initram} ${fdt_addr}
